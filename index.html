// ၁။ မင်းရဲ့ VPS IP ကို သတ်မှတ်ခြင်း const API_URL = "http://157.85.97.140:5000/create-ssh"; async function createVPN() { const userName = document.getElementById('vpn_username').value; const userPass = document.getElementById('vpn_password').value; // အချက်အလက် ပြည့်စုံမှု ရှိမရှိ စစ်ဆေးခြင်း if (!userName || !userPass) { alert("ကျေးဇူးပြု၍ Username နှင့် Password ထည့်ပါ။"); return; } try { // ၂။ VPS API ဆီသို့ Data ပို့ခြင်း const response = await fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: userName, password: userPass, days: 30 // သက်တမ်း ရက် ၃၀ သတ်မှတ်ချက် }) }); const result = await response.json(); if (result.status === "success") { alert("အောင်မြင်သည်- " + result.message); // Balance နှုတ်ခြင်း သို့မဟုတ် Dashboard Update လုပ်ခြင်း ကုဒ်များ ဤနေရာတွင် ဆက်ရေးနိုင်သည် } else { alert("မှားယွင်းမှုရှိသည်- " + result.message); } } catch (error) { console.error("Error:", error); alert("Server နှင့် ချိတ်ဆက်၍မရပါ။ API Run ထားခြင်း ရှိမရှိ ပြန်စစ်ပါ။"); } }
